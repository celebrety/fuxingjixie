# 复兴机械厂网站技术升级方案

## 一、 问题分析

### 1.1 数据同步问题
**现状：**
- `index.html`（客户访问页）和 `admin.html`（管理后台）都使用 `localStorage` 存储数据
- `localStorage` 是浏览器本地存储，仅限同一设备、同一浏览器访问
- 客户在手机访问 `index.html` 提交的留言存储在手机上
- 厂家在电脑访问 `admin.html` 读取的是电脑上的 `localStorage`
- **结果：** 两个设备的数据完全隔离，无法互相读取

**受影响功能：**
- 客户留言数据无法同步到管理后台
- 管理后台配置的图片无法同步到客户页面

### 1.2 H5移动端导航问题
**现状：**
- 底部导航栏使用 `tabClick()` 函数调用 `smoothTo()` 实现滚动
- `smoothTo()` 使用 `window.scrollTo()` 进行平滑滚动
- 但存在以下问题：
  - 可能被浏览器默认行为拦截
  - `event.preventDefault()` 和 `event.stopPropagation()` 使用不当
  - 在某些移动浏览器上兼容性差

## 二、 技术解决方案

### 2.1 数据同步方案 - 云端数据库 + API

#### 方案选择：LeanCloud 云服务（推荐）

**为什么选择 LeanCloud：**
- ✅ 免费额度充足（1000 API调用/天，1GB存储）
- ✅ 无需自建服务器，零运维成本
- ✅ 提供完整的 JavaScript SDK
- ✅ 支持实时数据同步
- ✅ 国内访问速度快
- ✅ 适合小型项目快速上线

**替代方案：**
1. **Firebase（Google）** - 国内访问不稳定
2. **Supabase** - 需要一定数据库知识
3. **自建 Node.js + MongoDB** - 需要服务器，运维成本高
4. **腾讯云开发 CloudBase** - 类似LeanCloud，也可考虑

### 2.2 H5导航修复方案

#### 问题分析

现有代码：
```javascript
function tabClick(el, section) {
    event.preventDefault();        // ❌ 问题1：event 未定义
    event.stopPropagation();       // ❌ 问题2：阻止冒泡可能影响其他逻辑
    document.querySelectorAll('.tab-item').forEach(t => t.classList.remove('active'));
    el.classList.add('active');
    smoothTo(section);
    return false;
}

function smoothTo(id) {
    const el = document.getElementById(id);
    if (!el) return;
    const topbar = document.getElementById('topbar');
    const offset = topbar ? topbar.offsetHeight : 56;
    window.scrollTo({ top: el.offsetTop - offset, behavior: 'smooth' });
}
```

**问题点：**
1. `event` 未作为参数传入，在某些浏览器中会报错
2. `window.scrollTo()` 在移动端可能被浏览器手势拦截
3. 没有考虑底部导航栏自身的高度偏移
4. 缺少降级方案（不支持 smooth scroll 的浏览器）

#### 修复方案

**方案1：改进 smoothTo 函数（推荐）**

```javascript
function smoothTo(id) {
    const el = document.getElementById(id);
    if (!el) return;
    
    // 计算偏移量（顶部导航 + 额外间距）
    const topbar = document.getElementById('topbar');
    const topbarHeight = topbar ? topbar.offsetHeight : 56;
    const extraOffset = 20; // 额外留白
    
    const targetPosition = el.offsetTop - topbarHeight - extraOffset;
    
    // 方案A：使用原生平滑滚动（现代浏览器）
    if ('scrollBehavior' in document.documentElement.style) {
        window.scrollTo({
            top: targetPosition,
            behavior: 'smooth'
        });
    } else {
        // 方案B：降级为动画滚动（兼容老浏览器）
        animateScroll(targetPosition, 500);
    }
}

// 动画滚动降级方案
function animateScroll(targetPosition, duration) {
    const startPosition = window.pageYOffset;
    const distance = targetPosition - startPosition;
    let startTime = null;
    
    function animation(currentTime) {
        if (startTime === null) startTime = currentTime;
        const timeElapsed = currentTime - startTime;
        const progress = Math.min(timeElapsed / duration, 1);
        
        // easeInOutCubic 缓动函数
        const ease = progress < 0.5 
            ? 4 * progress * progress * progress 
            : 1 - Math.pow(-2 * progress + 2, 3) / 2;
        
        window.scrollTo(0, startPosition + distance * ease);
        
        if (timeElapsed < duration) {
            requestAnimationFrame(animation);
        }
    }
    
    requestAnimationFrame(animation);
}
```

**方案2：修复 tabClick 函数**

```javascript
function tabClick(el, section, event) {
    // 正确接收 event 参数
    if (event) {
```


**方案3：使用 scrollIntoView API（最简单）**

```javascript
function smoothTo(id) {
    const el = document.getElementById(id);
    if (!el) return;
    
    // 使用原生 scrollIntoView API
    el.scrollIntoView({
        behavior: 'smooth',
        block: 'start'
    });
    
    // 修正偏移（滚动完成后微调）
    setTimeout(() => {
        const topbar = document.getElementById('topbar');
        const offset = topbar ? topbar.offsetHeight : 56;
        window.scrollBy(0, -offset - 20);
    }, 500);
}
```

#### 推荐综合方案

结合以上方案，最佳实践：

```javascript
/**
 * 平滑滚动到指定元素
 * @param {string} id - 目标元素ID
 */
function smoothTo(id) {
    const el = document.getElementById(id);
    if (!el) {
        console.warn('Element not found:', id);
        return;
    }
    
    // 计算目标位置
    const topbar = document.getElementById('topbar');
    const topbarHeight = topbar ? topbar.offsetHeight : 56;
    const extraOffset = 20;
    const targetPosition = el.offsetTop - topbarHeight - extraOffset;
    
    // 优先使用现代API
    if ('scrollBehavior' in document.documentElement.style) {
        window.scrollTo({
            top: targetPosition,
            behavior: 'smooth'
        });
    } else {
        // 降级方案：自定义动画
        animateScrollTo(targetPosition, 600);
    }
    
    // 更新URL hash（不触发跳转）
    if (history.pushState) {
        history.pushState(null, null, '#' + id);
    }
}

/**
 * 动画滚动（降级方案）
 */
function animateScrollTo(targetPosition, duration) {
    const startPosition = window.pageYOffset || document.documentElement.scrollTop;
    const distance = targetPosition - startPosition;
    let startTime = null;
    
    // easeInOutCubic 缓动函数
    function ease(t) {
        return t < 0.5 
            ? 4 * t * t * t 
            : 1 - Math.pow(-2 * t + 2, 3) / 2;
    }
    
    function step(currentTime) {
        if (startTime === null) startTime = currentTime;
        const elapsed = currentTime - startTime;
        const progress = Math.min(elapsed / duration, 1);
        
        window.scrollTo(0, startPosition + distance * ease(progress));
        
        if (progress < 1) {
            requestAnimationFrame(step);
        }
    }
    
    requestAnimationFrame(step);
}

/**
 * 底部导航点击处理
 */
function tabClick(el, section, event) {
    if (event) {
        event.preventDefault();
        event.stopPropagation();
    }
    
    // 更新激活状态
    document.querySelectorAll('.tab-item').forEach(tab => {
        tab.classList.remove('active');
    });
    el.classList.add('active');
    
    // 执行滚动
    smoothTo(section);
    
    return false;
}
```

```javascript
        await saveToCloud(formData);
        showSuccess('提交成功！');
    } catch (error) {
        console.warn('云端保存失败，使用本地缓存:', error);
        // 降级到本地缓存
        saveToLocal(formData);
        showSuccess('已本地保存，网络恢复后自动同步');
        // 设置定时同步
        scheduleSync();
    }
}

// 定时同步本地缓存
function scheduleSync() {
    const syncInterval = setInterval(async () => {
        if (navigator.onLine) {
            try {
                await syncLocalToCloud();
                clearInterval(syncInterval);
            } catch (e) {
                console.warn('同步失败，下次继续');
            }
        }
    }, 60000); // 每分钟尝试一次
}
```
